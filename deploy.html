Ops Guide

Build & Deploy

Multipass (dev) → Capistrano → Incus/LXD on Rocky Linux (production)

No Docker. Native Ruby + Postgres + Redis throughout.

01 Architecture

┌─────────────────────────────────────────────────────┐
│
│ Apple Silicon Mac — Dev
│ └── Multipass (Ubuntu 24.04 VM)
│ ├── Ruby 3.4.1 + Rails (via asdf)
│ ├── PostgreSQL 16
│ ├── Redis 7
│ └── foreman → Puma + Solid Queue + Tailwind
│
│ Production: Rocky Linux
│ └── Incus / LXD system containers
│ ├── app Ruby 3.4.1 + Rails + Puma
│ ├── worker Solid Queue worker
│ ├── postgres PostgreSQL 16
│ └── redis Redis 7
│
│ Deploy: Capistrano
│ └── git push → cap production deploy
│ ├── SSH into Incus app container
│ ├── git pull + bundle install
│ ├── assets:precompile + db:migrate
│ └── systemd restart puma + solid_queue
│
└─────────────────────────────────────────────────────┘

02 Mac Prerequisites

Install Multipass

brew install --cask multipass
# or download from https://multipass.run

Launch VM

multipass launch 24.04 \
  --name shopfloor-dev \
  --cpus 4 \
  --memory 8G \
  --disk 50G

multipass shell shopfloor-dev

Mount project

# On Mac, not inside VM:
multipass mount /path/to/your/project shopfloor-dev:/home/ubuntu/app

03 VM Bootstrap 

#!/bin/bash
# scripts/vm-bootstrap.sh — run once inside the Multipass VM
set -euo pipefail

echo "=== Packages ==="
sudo apt update && sudo apt upgrade -y
sudo apt install -y \
  curl git build-essential libssl-dev libreadline-dev \
  zlib1g-dev libyaml-dev libxml2-dev libxslt1-dev \
  libpq-dev postgresql postgresql-contrib redis-server \
  imagemagick libsqlite3-dev nginx

echo "=== asdf (Ruby + Node version manager) ==="
git clone https://github.com/asdf-vm/asdf.git ~/.asdf
echo '. "$HOME/.asdf/asdf.sh"' >> ~/.bashrc
echo '. "$HOME/.asdf/completions/asdf.bash"' >> ~/.bashrc
source ~/.bashrc

echo "=== Ruby 3.4.1 ==="
asdf plugin add ruby
asdf install ruby 3.4.1
asdf global ruby 3.4.1

echo "=== Node.js 22 ==="
asdf plugin add nodejs
asdf install nodejs 22.14.0
asdf global nodejs 22.14.0

echo "=== Rails + Capistrano ==="
gem install rails bundler
gem install capistrano capistrano-rails capistrano-bundler capistrano-rvm

echo "=== Postgres dev user ==="
sudo -u postgres createuser -s ubuntu 2>/dev/null || true

echo "=== Done ==="
echo "Log out and back in: exit → multipass shell shopfloor-dev"
echo "Then: cd ~/app && bundle install && rails db:create db:migrate"

05 Capistrano Setup

Gemfile

group :development do
  gem "capistrano", "~> 3.19"
  gem "capistrano-rails", "~> 1.7"
  gem "capistrano-bundler", "~> 2.1"
  gem "capistrano-rvm"
end

Capfile

# Capfile
require "capistrano/setup"
require "capistrano/deploy"
require "capistrano/rails"
require "capistrano/bundler"
require "capistrano/rvm"

install_plugin Capistrano::Puma
install_plugin Capistrano::SolidQueue

Dir.glob("lib/capistrano/tasks/*.rake").each { |r| import r }

config/deploy.rb

# config/deploy.rb
lock "~> 3.19"

set :application, "shopfloor"
set :repo_url, "git@github.com:your-org/shopfloor.git"
set :branch, "main"

set :deploy_to, "/home/deploy/shopfloor"
set :keep_releases, 5

set :rvm_ruby_version, "3.4.1"

set :linked_files, %w[config/master.key config/database.yml]
set :linked_dirs, %w[log tmp/pids tmp/cache tmp/sockets storage public/uploads]

set :puma_service_unit_name, "puma"
set :puma_systemctl_user, :system

namespace :deploy do
  after :finishing, "puma:restart"
  after :finishing, "solid_queue:restart"
end

config/deploy/production.rb

# config/deploy/production.rb
server "10.0.0.50", user: "deploy", roles: %w[app web db]

set :branch, "main"
set :rails_env, "production"

Deploy commands

# First deploy (provision directories)
cap production deploy:check

# Deploy
cap production deploy

# Rollback
cap production deploy:rollback

# Run Rails console remotely
cap production rails:console

06 Production Host Setup

Rocky Linux — System Containers

# Install Incus (community fork of LXD)
sudo dnf install -y epel-release
sudo dnf install -y incus incus-client
sudo systemctl enable --now incus
sudo usermod -aG incus $USER

# Or LXD via snap:
# sudo dnf install -y snapd
# sudo snap install lxd

scripts/provision-incus-containers.sh

#!/bin/bash
# Run on the Rocky Linux host to create production system containers
set -euo pipefail

INCUS="incus"   # or "lxc" if using LXD

echo "=== Launching Postgres container ==="
$INCUS launch images:rockylinux/9/cloud postgres
$INCUS exec postgres -- dnf install -y postgresql-server
$INCUS exec postgres -- postgresql-setup --initdb
$INCUS exec postgres -- systemctl enable --now postgresql
$INCUS config device add postgres data disk source=/data/postgres path=/var/lib/pgsql/data

echo "=== Launching Redis container ==="
$INCUS launch images:rockylinux/9/cloud redis
$INCUS exec redis -- dnf install -y redis
$INCUS exec redis -- systemctl enable --now redis

echo "=== Launching App container ==="
$INCUS launch images:rockylinux/9/cloud app
$INCUS config device add app postgres disk source=postgres path=/dev  # shared network

# Inside the app container:
$INCUS exec app -- dnf install -y \
  git curl gcc make libssl-devel libreadline-devel \
  zlib-devel libyaml-devel libxml2-devel libxslt-devel \
  postgresql-devel postgresql redis imagemagick sqlite-devel

# Create deploy user
$INCUS exec app -- useradd -m -s /bin/bash deploy
$INCUS exec app -- mkdir -p /home/deploy/shopfloor
$INCUS exec app -- chown deploy:deploy /home/deploy/shopfloor

echo "=== Apply firewall rules ==="
firewall-cmd --permanent --add-port=443/tcp --add-port=80/tcp
firewall-cmd --reload

echo "=== Done ==="

Systemd services inside the app container

# /etc/systemd/system/puma.service
[Unit]
Description=Puma Rails Server
After=network.target

[Service]
Type=simple
User=deploy
WorkingDirectory=/home/deploy/shopfloor/current
ExecStart=/home/deploy/.asdf/shims/bundle exec puma -C config/puma.rb
ExecReload=/bin/kill -USR1 $MAINPID
Restart=always

[Install]
WantedBy=multi-user.target

07 Weigh Station (Ubuntu Box)

Each weigh station runs Ubuntu with a browser pointing at the Rails app. The scale (RS-232/USB HID), USB camera, Zebra printer, and NFC reader connect locally.

#!/bin/bash
# scripts/setup-weigh-station.sh — run on each Ubuntu station box
set -euo pipefail

echo "=== Printer drivers ==="
sudo apt install -y cups printer-driver-zebra
sudo systemctl enable --now cups

echo "=== USB scale permissions ==="
echo 'SUBSYSTEM=="usb", ATTRS{idVendor}=="*", ATTRS{idProduct}=="*", MODE="0666"' \
  | sudo tee /etc/udev/rules.d/99-scale.rules
sudo udevadm control --reload-rules && sudo udevadm trigger

echo "=== Configure printer ==="
echo "Open http://localhost:631 (CUPS) to add Zebra printer"
echo "Or find USB URI with: lpinfo -v | grep usb"
echo "Then: sudo lpadmin -p Zebra -E -v usb://... -m drv:///..."

08 CI Pipeline 

# .github/workflows/ci.yml
name: CI

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: ruby/setup-ruby@v1
        with:
          bundler-cache: true
      - run: bundle exec standardrb
      - run: bundle exec brakeman --no-pager
      - run: bundle exec bundler-audit check --update

  test:
    runs-on: ubuntu-latest
    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_USER: runner
          POSTGRES_DB: shopfloor_test
          POSTGRES_PASSWORD: ""
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
      redis:
        image: redis:7
        options: >-
          --health-cmd "redis-cli ping"
          --health-interval 10s
    steps:
      - uses: actions/checkout@v4
      - uses: ruby/setup-ruby@v1
        with:
          bundler-cache: true
      - run: |
          cp config/database.yml.ci config/database.yml
          bundle exec rails db:create db:migrate
          bundle exec rspec

  deploy:
    needs: [lint, test]
    if: github.ref == 'refs/heads/main'
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: ruby/setup-ruby@v1
        with:
          bundler-cache: true
      - name: Deploy via Capistrano
        env:
          SSH_PRIVATE_KEY: ${{ secrets.SSH_PRIVATE_KEY }}
          RAILS_MASTER_KEY: ${{ secrets.RAILS_MASTER_KEY }}
        run: |
          eval "$(ssh-agent -s)"
          echo "$SSH_PRIVATE_KEY" | ssh-add -
          bundle exec cap production deploy
